export default class App extends Component {
  constructor(){
    super();
    this.state = {
      name: 'Kürşad KARAGÖZ',
      year: 1996,
    }
  }

  changeState = () => {
    this.setState({
      name: 'Ali Veli',
      year: 1990
    },() => {
      alert('Bilgiler Güncellendi.')
    })
  }

  render(){
    const {name, year} = this.state;
    return (
      <View style={[{flex:1, paddingTop:150}]}>
        <Text> {name} {year} </Text>
        <TouchableOpacity onPress={this.changeState}>
          <Text> Değiştir</Text>
        </TouchableOpacity>
        
      </View>
    )
    
  }
}
